# MapComp: Map Component


import { MapComp, RasterLayer, VectorLayer } from "../modules/myMap/myMap.js"

<MapComp height="600px">
    <VectorLayer 
      name="Sites"
      path2geojson="../data/elamortuary.geojson"
      fitToContent={true}
      checked={true}
      popupTemplate={
      (attribs) => { return `<p><strong>${attribs.id}</strong> ${attribs.site}`; }
      }
      pointToLayer={
        (f, ll) => {
          return L.circleMarker(ll, {
            radius: 4,
            fillColor: "#FE04FC",
            color: "#000",
            weight: 1,
            opacity: 1,
            fillOpacity: 0.8
          });
        }}
    />

    <RasterLayer 
      name="Open Street Map"
      url = "https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png"
      attribution= '&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> contributors' />
    
    <RasterLayer 
      name="Esri Imagery/Satellite"
      url = "https://server.arcgisonline.com/ArcGIS/rest/services/World_Imagery/MapServer/tile/{z}/{y}/{x}"
      checked={true} />
    
    <RasterLayer 
      name="Esri Streets"
      url = "https://server.arcgisonline.com/ArcGIS/rest/services/World_Street_Map/MapServer/tile/{z}/{y}/{x}" />
    
  </MapComp>
